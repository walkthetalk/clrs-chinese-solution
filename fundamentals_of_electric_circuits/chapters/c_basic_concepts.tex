% 1.1
\startQ
How much charge is represented by these number of electrons?
\startIG
\item $6.482 \times 10^{17}$
\item $1.24 \times 10^{18}$
\item $2.46 \times 10^{19}$
\item $1.628 \times 10^{20}$
\stopIG
\stopQ

\startA
\startIG
\item $q = 6.482 \times 10^{17} \times (1.602\times 10^{-19} \Coulomb) = 0.10384 \Coulomb$
\item $q = 1.24  \times 10^{18} \times (1.602\times 10^{-19} \Coulomb) = 0.19865 \Coulomb$
\item $q = 2.46  \times 10^{19} \times (1.602\times 10^{-19} \Coulomb) = 3.941   \Coulomb$
\item $q = 1.628 \times 10^{20} \times (1.602\times 10^{-19} \Coulomb) = 26.08   \Coulomb$
\stopIG
\stopA
