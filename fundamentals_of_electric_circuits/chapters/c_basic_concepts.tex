% 1.1
\startQ
How much charge is represented by these number of electrons?
\startIG
\item $\unit{6.482e17}$
\item $\unit{1.24=e18}$
\item $\unit{2.46=e19}$
\item $\unit{1.628e20}$
\stopIG
\stopQ

\startA
\startIG
\item $q = \unit{6.482e17} \times (\unit{1.602e-19 coulomb}) = \unit{_0.10384 coulomb}$
\item $q = \unit{1.24=e18} \times (\unit{1.602e-19 coulomb}) = \unit{_0.19865 coulomb}$
\item $q = \unit{2.46=e19} \times (\unit{1.602e-19 coulomb}) = \unit{_3.941== coulomb}$
\item $q = \unit{1.628e20} \times (\unit{1.602e-19 coulomb}) = \unit{26.08=== coulomb}$
\stopIG
\stopA
