\startluacode
function commands.splitwords(str)
	words = string.split(str, ' ')
	groupsize = 4
	for i=1,#words,groupsize do
		tex.sprint('\\startxrow[frame=on]')
		tex.sprint('\\startxcell')
		for k = 0,3 do
			idx = i + k
			if (idx > #words) then
				break
			end
			chars = utf.split(words[idx])
			for j=1,#chars do
				tex.sprint(chars[j])
			end
			if (i ~= #words) and (k ~= 3) then
				tex.sprint('\\kern.5em')
			end
		end
		tex.sprint('\\stopxcell')
		tex.sprint('\\stopxrow')
	end
end
\stopluacode
